# Holy Site PM Events I
eocfm.0010 = { # Rome Holy Site Event
	type = country_event
	hidden = yes
	
	trigger = {
		has_state_in_state_region = STATE_LAZIO
		OR = {
			AND = {
				has_variable = eocfm_controls_rome
				s:STATE_LAZIO = {
					any_scope_state = {
						owner = root
						is_production_method_active = {
							building_type = eocfm_building_patriarchate_rome
							production_method = pm_eocfm_patriarchate_occupied
						}
					}
				}
			}
			AND = {
				NOT = {
					has_variable = eocfm_controls_rome
				}
				s:STATE_LAZIO = {
					any_scope_state = {
						owner = root
						is_production_method_active = {
							building_type = eocfm_building_patriarchate_rome
							production_method = pm_eocfm_patriarchate_rome
						}
					}
				}
			}
		}
	}
	
	immediate = {
		if = {
			limit = {
				has_variable = eocfm_controls_rome
			}
			s:STATE_LAZIO = {
				random_scope_state = {
					limit = {
						owner = root
					}
					activate_production_method = {
						building_type = eocfm_building_patriarchate_rome
						production_method = pm_eocfm_patriarchate_rome
					}
				}
			}
		}
		else_if = {
			limit = {
				NOT = {
					has_variable = eocfm_controls_rome
				}
			}
			s:STATE_LAZIO = {
				random_scope_state = {
					limit = {
						owner = root
					}
					activate_production_method = {
						building_type = eocfm_building_patriarchate_rome

				}

		}
	}
