eocfm_rome_opinion_value = {
	value = 0

		}

			value = 10
			desc = eocfm_value_from_matching_religion
		}
	}

		}
		add = {
			value = 20

		}
	}

		limit = {

		}

		}
	}
	if = {
		limit = {
			has_variable = eocfm_controls_rome
		}
		add = {
			value = 10
			desc = eocfm_value_from_seat_control
		}
	}
	if = {
		limit = {
			var:eocfm_liturgical_rite_var = global_var:eocfm_rome_liturgical_rite_var
		}
		add = {
			value = 10
			desc = eocfm_value_from_matching_liturgial_rite
		}
	}
	
	# Mixed
	add = {
		value = var:eocfm_rome_relations_var
		desc = eocfm_value_from_relations
	}
	
	#Penalities
	if = {
		limit = {

		}
		add = {
			value = -20
			desc = eocfm_value_from_different_religion
		}
	}
	if = {
		limit = {
			has_law = law_type:law_total_separation
		}
		add = {
			value = -10
			desc = eocfm_value_from_total_separation
		}
	}

			has_law = law_type:law_state_atheism
		}

			desc = eocfm_value_from_state_atheism
