## MAJOR SEAT INTERACTIONS
eocfm_declare_autocephaly_sgui = {
	scope = country
	
	is_shown = {
		NOR = {
			has_variable = eocfm_is_autocephalous
			has_variable = eocfm_schism_var
		}
	}
	
	is_valid = {
		is_subject = no
		eocfm_has_500_caesaropapism_trigger = yes
		NOR = {
			has_law = law_type:law_theocracy
			has_modifier = modifier_eocfm_autocephaly_abandoned
		}
		trigger_if = {
			limit = {
				OR = {
					c:MOL ?= this
					c:WAL ?= this
				}
			}
			custom_tooltip = {
				text = eocfm_romania_united
				hidden_trigger = {
					c:ROM ?= this
				}
			}
		}
	}
	
	effect = {
		set_variable = {
			name = eocfm_schism_var
			value = 0
		}
		add_modifier = {
			name = modifier_eocfm_schismatic
			months = 120
		}
		hidden_effect = {
			if = {
				limit = {
					eocfm_is_roman_catholic_trigger = yes
				}
				trigger_event = {
					id = eocfm.0100 # Declaration of Schism with Rome
					popup = yes
				}
			}
			else_if = {
				limit = {
					eocfm_is_eastern_christian_trigger = yes
				}
				trigger_event = {
					id = eocfm.0101 # Declaration of Schism with Constaintinople
					popup = yes
				}
			}
			else_if = {
				limit = {
					eocfm_is_oriental_orthodox_trigger = yes
				}
				trigger_event = {
					id = eocfm.0102 # Declaration of Schism with Coptic Alexandria
					popup = yes
				}
			}
			else_if = {
				limit = {
					eocfm_is_elysian_orthodox_trigger = yes
				}
				trigger_event = {
					id = eocfm.0103 # Declaration of Schism with Elysium
					popup = yes
				}
			}
		}
	}
}
